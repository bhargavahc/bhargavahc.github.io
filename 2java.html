    public void add(View v)

    public void divide(View v)
    {

    }
    public void compute(View v)
    {
        float n1,n2,r;
        switch (op)
        {
            case '+':n1=Float.parseFloat(number1.getText().toString());
                     n2=Float.parseFloat(number2.getText().toString());
                     r=n1+n2;
                     res.setText(""+n1+"+"+n2+"="+r);
                     break;
            case '-':n1=Float.parseFloat(number1.getText().toString());
                     n2=Float.parseFloat(number2.getText().toString());
                     r=n1-n2;
                     res.setText(""+n1+"-"+n2+"="+r);
                     break;
            case '*':n1=Float.parseFloat(number1.getText().toString());
                     n2=Float.parseFloat(number2.getText().toString());
                     r=n1*n2;
                     res.setText(""+n1+"*"+n2+"="+r);
                     break;
            case '/':n1=Float.parseFloat(number1.getText().toString());
                     n2=Float.parseFloat(number2.getText().toString());
                     r=n1/n2;
                     res.setText(""+n1+"/"+n2+"="+r);
                     break;
        }
    }
    public void clear(View v)
    {

    public void clear_one_field_completely(View v)
    {

        {

        }

        }
    }
    public void clear_one_digit_at_a_time(View v)
    {
        if(number1.hasFocus())
        {
            String n;
            n=number1.getText().toString();
            n=n.substring(0,n.length()-1);

            number1.setText(n);
        }
        else if(number2.hasFocus())
        {
            String n;
            n=number2.getText().toString();
            n=n.substring(0,n.length()-1);
            number2.setText(n);
        }
        else
        {
            Toast.makeText(this, "Please click on Number1/Number2 Field", Toast.LENGTH_LONG).show();
        }
    }
